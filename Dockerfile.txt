# Use official Python image
FROM python:3.11-slim

# Install deps required by playwright
RUN apt-get update && apt-get install -y wget ca-certificates libnss3 libatk1.0-0 libgtk-3-0 libx11-xcb1 libxcomposite1 libxdamage1 libxrandr2 libgbm-dev libasound2 libpangocairo-1.0-0 libxrender1 libdrm2 libxshmfence1 libxcb-shm0 libxcb1 libgtk-3-0 curl unzip && \
    rm -rf /var/lib/apt/lists/*

WORKDIR /app
COPY requirements.txt /app/requirements.txt
RUN pip install --upgrade pip
RUN pip install -r requirements.txt

# Install Playwright browsers
RUN playwright install --with-deps

COPY . /app

ENV PYTHONUNBUFFERED=1

CMD ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "8000"]
